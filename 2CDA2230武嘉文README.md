limit = int(input("いくつまで数えますか？: "))


for num in range(1, limit + 1):
    if num % 15 == 0:
        print("Fizz Buzz")
    elif num % 3 == 0:
        print("Fizz")
    elif num % 5 == 0:
        print("Buzz")
    else:
        print(num)
